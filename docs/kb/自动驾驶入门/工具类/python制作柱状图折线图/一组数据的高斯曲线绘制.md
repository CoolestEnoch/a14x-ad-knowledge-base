```python
import numpy as np
import matplotlib.pyplot as plt
import scipy.stats as stats
import pickle

# 字体设置
xy_label_font = 40
stick_font = 40
lenged_font = 40
title_font = 50

# 离线的数据读取
with open('train.pkl', 'rb') as f:
    train_pixel_mean_list = pickle.load(f)

with open('val.pkl', 'rb') as f:
    val_pixel_mean_list = pickle.load(f)

with open('val_cor.pkl', 'rb') as f:
    cor_pixel_mean_list = pickle.load(f)

data_train = np.array(train_pixel_mean_list)
data_val = np.array(val_pixel_mean_list)
data_cor = np.array(cor_pixel_mean_list)

# 计算均值和方差
mean_value_train = np.mean(data_train)
variance_value_train = np.var(data_train)
mean_value_val = np.mean(data_val)
variance_value_val = np.var(data_val)
mean_value_cor = np.mean(data_cor)
variance_value_cor = np.var(data_cor)

plt.figure(figsize=(16, 16))
# ax1.set_xlabel('X', fontsize=xy_label_font)  # 设置x轴标签的字体大小为12
# ax1.set_ylabel('Y', fontsize=xy_label_font)  # 设置y轴标签的字体大小为12
# plt.title('Data Distribution', fontsize=title_font)  # 设置子图标题的字体大小为14

# 绘制方差高斯线
x_train = np.linspace(np.min(data_train) - np.min(data_train) - 1, np.max(data_train), 1050)
pdf_train = stats.norm.pdf(x_train, mean_value_train, np.sqrt(variance_value_train))
plt.plot(x_train, pdf_train, linewidth=2, label='Clean Train')

x_val = np.linspace(np.min(data_val) - np.min(data_val) - 1, np.max(data_val), 1050)
pdf_val = stats.norm.pdf(x_val, mean_value_val, np.sqrt(variance_value_val))
plt.plot(x_val, pdf_val, linewidth=2, label='Clean Val')

x_cor = np.linspace(np.min(data_cor) - np.min(data_cor) - 1, np.max(data_cor), 1050)
pdf_cor = stats.norm.pdf(x_cor, mean_value_cor, np.sqrt(variance_value_cor))
plt.plot(x_cor, pdf_cor, color='gray', linewidth=2, label='Noise Set')

# 绘制阴影
plt.fill_between(x_train, pdf_train, alpha=0.3)
plt.fill_between(x_val, pdf_val, alpha=0.3) 
plt.fill_between(x_cor, pdf_cor, color='gray', alpha=0.3)
# plt.legend(fontsize=lenged_font)
plt.xlim(45, 170)
plt.ylim(0, 0.07, 5)
plt.tick_params(axis='both', which='major', labelsize=stick_font)  # 设置刻度标签的字体大小为10)

# ax2.set_xlabel('X', fontsize=xy_label_font)  # 设置x轴标签的字体大小为12
# ax2.set_ylabel('Y', fontsize=xy_label_font)  # 设置y轴标签的字体大小为12


# 设置图例和标题
plt.savefig('variance.png', dpi=300)
plt.show()
```
